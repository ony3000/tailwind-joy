# Switch

<AvailableFrom version="0.3.0" />

Switches toggle the state of a single setting on or off.

## Basics

```jsx
import { Switch } from 'tailwind-joy/components';
```

## Customization

### Controlled

To create a controlled switch, use the `checked` and `onChange` props.

export function SwitchControlled() {
  const [checked, setChecked] = useState(false);
  return (
    <DisplayStand>
      <CssVarsProvider modeStorageKey="theme">
        <JoySwitch
          checked={checked}
          onChange={(e) => setChecked(e.currentTarget.checked)}
        />
      </CssVarsProvider>
    </DisplayStand>
  );
}

<SwitchControlled />

### Decorators

To insert icon decorators, use the `startDecorator` and/or `endDecorator` props.

export function SwitchDecorators() {
  const [checked, setChecked] = useState(false);
  return (
    <DisplayStand>
      <CssVarsProvider modeStorageKey="theme">
        <JoySwitch
          color={checked ? 'primary' : 'danger'}
          startDecorator={
            <LocalFireDepartmentRoundedIcon
              sx={[
                checked ? { color: 'text.tertiary' } : { color: 'danger.600' },
              ]}
            />
          }
          endDecorator={
            <WavesRoundedIcon
              sx={[
                checked ? { color: 'primary.500' } : { color: 'text.tertiary' },
              ]}
            />
          }
          checked={checked}
          onChange={(e) => setChecked(e.currentTarget.checked)}
        />
      </CssVarsProvider>
    </DisplayStand>
  );
}

<SwitchDecorators />
